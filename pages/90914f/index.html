训练自己模型

# 训练自己模型

# 准备数据集

我使用的为COCO格式的数据集，自行按照其对应的格式准备。

# 准备配置文件

假如我使用CASCADE_RCNN，如下为我们的配置文件，将其命名为cascade_rcnn_r50_fpn_20e_coco.py，保存到configs/cascade_rcnn，其内容如下：

_base_ = './cascade_rcnn_r50_fpn_20e_coco.py'
model = dict(
    type='CascadeRCNN',
    backbone=dict(
        type='ResNeXt',
        depth=101,
        groups=64,
        base_width=4,
        num_stages=4,
        out_indices=(0, 1, 2, 3),
        frozen_stages=1,
        norm_cfg=dict(type='BN', requires_grad=True),
        style='pytorch',
        init_cfg=dict(
            type='Pretrained', checkpoint='open-mmlab://resnext101_64x4d')))

# 数据集类型
dataset_type = 'COCODataset'
# 类别
classes = ('sy', 'gy', 'lk')
# 数据集
data = dict(
    # 训练集
    train=dict(
        img_prefix='/home/lgh/mmlabTest/data/coco/train/',
        classes=classes,
        ann_file='/home/lgh/mmlabTest/data/coco/train.json'),
    # 验证集
    val=dict(
        img_prefix='/home/lgh/mmlabTest/data/coco/train/',
        classes=classes,
        ann_file='/home/lgh/mmlabTest/data/coco/val.json'),
    # 测试集
    test=dict(
        img_prefix='/home/lgh/mmlabTest/data/coco/train/',
        classes=classes,
        ann_file='/home/lgh/mmlabTest/data/coco/test.json'))

# 每多少个epoch测试一次
checkpoint_config = dict(interval=10)
# 梯度裁减
optimizer_config = dict(grad_clip=dict(max_norm=35, norm_type=2))
